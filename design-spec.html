10. Tailwind CSS 类映射表

/* 常用类和对应的 CSS 值 */ /* 间距 */ - px-4 = padding-left/right: 1rem (16px) - py-5 = padding-top/bottom: 1.25rem (20px) - py-6 = padding-top/bottom: 1.5rem (24px) - gap-8 = gap: 2rem (32px) /* 宽高 */ - w-20 = width: 5rem (80px) - h-20 = height: 5rem (80px) - w-28 = width: 7rem (112px) - h-28 = height: 7rem (112px) /* 背景颜色 */ - bg-black = background-color: #000000 - bg-white = background-color: #FFFFFF - bg-red-500 = background-color: #EF4444 - bg-gray-300 = background-color: #D1D5DB /* 圆角 */ - rounded-lg = border-radius: 0.5rem (8px) - rounded-2xl = border-radius: 1rem (16px) - rounded-3xl = border-radius: 1.5rem (24px) - rounded-full = border-radius: 9999px (完全圆形) /* 阴影 */ - shadow-lg = box-shadow: 0 10px 15px -3px rgba(0,0,0,0.1)
